function gcd_loop(m, n)
    m = copy(m)
    n = copy(n)

    if m < n
        m, n = n, m
    end

    global count_div = 0

    while n > 0
        r = m % n
        m, n = n, r

        count_div += 1
    end

    return m
end

function fibonacci(n)
    if n <= 0
        return 0
    elseif n == 1
        return 1
    else
        a, b = Int128(0), Int128(1)
        for i in 2:n
            a, b = b, a + b
        end
        return b
    end
end
